SOME GGPLOTS

ggplot(dat, aes(x=age_at_booking, fill=factor(race))) +
    geom_density(alpha=.5) + 
    facet_grid(race ~ .)+
    xlab("Age at booking") + 
    ylab("Density by race") +
    theme(plot.title = element_text(size = 20)) +
    labs(title='Age at booking summary\nby race\n')
